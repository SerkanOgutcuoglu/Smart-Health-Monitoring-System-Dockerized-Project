CREATE DATABASE IF NOT EXISTS healthdb;

USE healthdb;

CREATE TABLE IF NOT EXISTS health_data (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    heart_rate INT,
    blood_pressure VARCHAR(20),
    temperature FLOAT,
    recorded_at DATETIME
);
